## API

```bash
npm i @k03mad/ip2geo
```

```js
import {ip2geo} from '@k03mad/ip2geo';

const info = await ip2geo('1.1.1.1', {
    // defaults
    cacheDir: path.join(os.tmpdir(), '.ip2geo'),
    cacheFileName: 'ips.log',
    cacheFileSeparator: ';;',
    cacheFileNewline: '\n',
});

// info {
//   ip: '1.1.1.1',
```
